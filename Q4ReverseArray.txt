//count of Negative Numbers in Array
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cout<<"Enter size of Array :";
    cin>>n;
    
    int a[n];
    
    cout<<"Enter elements in array :" ;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    
    //only print in reverse
    for(int i=n-1;i>=0;i--)
    {
        cout<<a[i]<<" ";
    }
   
   // Reverse Position in Array
   int start=0,end=n-1;
   while(start<end)
   {
       swap(a[start],a[end]);
       start++,end--;
   }
   
    for(int i=0;i<n;i++)
    {
        cout<<a[i]<<" ";
    }
    
    return 0;
}
