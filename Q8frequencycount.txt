//frequency count
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cout<<"Enter size of Array :";
    cin>>n;
    
    int a[n];
    
   map<int,int>mp;

    cout<<"Enter elements in array :" ;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    
     for(int i=0;i<n;i++)
    {
        mp[a[i]]++;
    }
    
    
   for(auto x : mp)
   {
       cout<<x.first <<" "<<x.second <<endl;
   }
 
    return 0;
}
