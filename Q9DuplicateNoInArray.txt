#include <bits/stdc++.h>
using namespace std;

int main()
{
     int n;
    cout<<"Enter size of Array :";
    cin>>n;
    
    map<int,int> mp;
    
    int a[n];
    
    cout<<"Enter elements in array :" ;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }

    for(int i=0;i<n;i++)
    {
       mp[a[i]]++;
    }

    cout<<"Duplicate Number is : ";
    for(auto x : mp)
    {
        if(x.second > 1)
        {
            cout<<x.first<<" ";
        }
    }

    return 0;
}
